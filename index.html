Click Deploy // ============================================ function doPost(e) { try { // Handle data from either fetch (postData) or form submission (parameter) var raw; if (e.postData && e.postData.contents) { raw = e.postData.contents; } else if (e.parameter && e.parameter.payload) { raw = e.parameter.payload; } else { // Try all parameters raw = JSON.stringify(e.parameter || {}); } var data = JSON.parse(raw); var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheetName = data.surveyType === "Teacher" ? "Teacher Responses" : "Student Responses"; var sheet = ss.getSheetByName(sheetName); if (!sheet) { sheet = ss.insertSheet(sheetName); if (data.surveyType === "Teacher") { sheet.getRange(1, 1, 1, 11).setValues([[ "Timestamp", "Name", "Pressure Demo vs Whiteboard", "Force Effects Rating (0-4)", "Teaching Scenario Choice", "Feature Value Ratings", "Quiz Assessment", "Teaching Methods Ranking", "Deck vs Video Comparison", "Would Adopt", "Open Feedback" ]]); } else { sheet.getRange(1, 1, 1, 12).setValues([[ "Timestamp", "Name", "Force Demo vs Video", "Pressure Understanding (0-4)", "Atmospheric Reaction", "Knowledge Check Score", "Experiment Ranking", "What Worked (multi)", "Confidence Before (1-10)", "Confidence After (1-10)", "Lab vs Video Preference", "Improvement + Want More" ]]); } sheet.getRange(1, 1, 1, sheet.getLastColumn()).setFontWeight("bold"); sheet.setFrozenRows(1); } var row = [new Date().toLocaleString()]; var answers = data.answers; var steps = data.steps; steps.forEach(function(stepId) { var val = answers[stepId]; if (val === undefined || val === null) { row.push(""); } else if (typeof val === "string" || typeof val === "number") { row.push(String(val)); } else if (Array.isArray(val)) { row.push(val.join(", ")); } else if (typeof val === "object") { // Make objects readable var parts = []; for (var k in val) { parts.push(k + ": " + val[k]); } row.push(parts.join(" | ")); } else { row.push(String(val)); } }); sheet.appendRow(row); return ContentService .createTextOutput(JSON.stringify({ status: "success" })) .setMimeType(ContentService.MimeType.JSON); } catch (err) { // Log the error for debugging Logger.log("Error: " + err.toString()); Logger.log("PostData: " + JSON.stringify(e.postData)); Logger.log("Parameter: " + JSON.stringify(e.parameter)); return ContentService .createTextOutput(JSON.stringify({ status: "error", message: err.toString() })) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { return ContentService .createTextOutput(JSON.stringify({ status: "ok", message: "Forces Survey API is running!" })) .setMimeType(ContentService.MimeType.JSON); }
